SELECT I.SUMMARY AS ISSUE, I.DESCRIPTION, IL.NAME AS ISSUESLIST
FROM ISSUES I
JOIN ISSUESLISTS IL ON I.ISSUESLISTS_ID = IL.ID;

SELECT I.SUMMARY AS ISSUE, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_AMOUNT
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) IS NOT NULL;
